suma = 0
while True:
    n = input('Vamos a multiplicar, dame un número: ')
    try:
        n = int(n)
    except ValueError:
        print('Escribe un número entero')
        continue
    if n <= 0:
        print('Ingresa un número entero mayor que 0')
        continue
    break
for i in range (1, 11):
    resultado = n * i
    print(f'{n} * {i} = {resultado}')
    suma += resultado
print(f'Suma acumulado de las tablas: {suma}')
